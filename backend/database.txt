CREATE TABLE artworks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    media_type TEXT NOT NULL, -- e.g., 'image', 'video', '3D_model', 'poetry'
    source TEXT NOT NULL, -- e.g., 'Flickr', 'YouTube', 'Vimeo'
    creator_name TEXT, -- Name of the original artist
    url TEXT NOT NULL UNIQUE, -- URL to the artwork
    title TEXT, -- Title of the artwork
    description TEXT, -- Optional description
    created_at TIMESTAMP DEFAULT NOW()
);
